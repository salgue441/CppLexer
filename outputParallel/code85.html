C#ProgramtoDisplayAllthePrimeNumbersBetween1to100usingSystem;usingSystem.Collections.Generic;usingSystem.Linq;usingSystem.Text;namespacePrimeNumber{
  classProgram{
    staticvoidMain(string[]args){
      boolisPrime=true;Console.WriteLine("PrimeNumbers:");for(inti=2;i<=100;i++){
        for(intj=2;j<=100;j++){
          if(i!=j&&i%j==0){
            isPrime=false;break;}
          }
        if(isPrime){
          Console.Write("\t"+i);}
        isPrime=true;}
      Console.ReadKey();}
    }
  }
