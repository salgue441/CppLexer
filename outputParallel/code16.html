C#ProgramtoFindtheSumoftwoBinaryNumbersusingSystem;usingSystem.Collections.Generic;usingSystem.Linq;usingSystem.Text;namespaceConsoleApplication{
  classProgram{
    staticvoidMain(string[]args){
      intb1,b2;inti=0,rem=0;int[]sum=newint[20];Console.WriteLine("Enterthefirstbinarynumber:");b1=int.Parse(Console.ReadLine());Console.WriteLine("Enterthesecondbinarynumber:");b2=int.Parse(Console.ReadLine());while(b1!=0||b2!=0){
        sum[i++]=(b1%10+b2%10+rem)%2;rem=(b1%10+b2%10+rem)/2;b1=b1/10;b2=b2/10;}
      if(rem!=0)sum[i++]=rem;--i;Console.WriteLine("Sumoftwobinarynumbers:");while(i>=0)Console.Write("{0}",sum[i--]);Console.ReadLine();}
    }
  }
